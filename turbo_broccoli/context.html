 15def _list_of_types_to_dict(lot: list[type]) -> dict[str, type]:
 16    """
 17    Converts a list of types `[T1, T2, ...]` to a dict that looks like `{"T1":
 18    T1, "T2": T2, ...}`.
 19    """
 20    return {t.__name__: t for t in lot}

 25    """
 26    (De)Serialization context, which is an object that contains various
 27    information and parameters about the ongoing operation. If you want your
 28    (de)serialization to behave a certain way, create a context object and pass
 29    it to
 30    [`turbo_broccoli.to_json`](https://altaris.github.io/turbo-broccoli/turbo_broccoli/turbo_broccoli.html#to_json)
 31    or
 32    [`turbo_broccoli.from_json`](https://altaris.github.io/turbo-broccoli/turbo_broccoli/turbo_broccoli.html#from_json).
 33    For convenience,
 34    [`turbo_broccoli.save_json`](https://altaris.github.io/turbo-broccoli/turbo_broccoli/turbo_broccoli.html#save_json)
 35    and
 36    [`turbo_broccoli.load_json`](https://altaris.github.io/turbo-broccoli/turbo_broccoli/turbo_broccoli.html#load_json)
 37    take the context parameter's as kwargs.
 38    """

 53    def __init__(
 54        self,
 55        file_path: str | Path | None = None,
 56        artifact_path: str | Path | None = None,
 57        min_artifact_size: int | None = None,
 58        nodecode_types: list[str] | None = None,
 59        keras_format: Literal["keras", "tf", "h5"] | None = None,
 60        pandas_format: (
 61            Literal[
 62                "csv",
 63                "excel",
 64                "feather",
 65                "html",
 66                "json",
 67                "latex",
 68                "orc",
 69                "parquet",
 70                "pickle",
 71                "sql",
 72                "stata",
 73                "xml",
 74            ]
 75            | None
 76        ) = None,
 77        pandas_kwargs: dict | None = None,
 78        nacl_shared_key: bytes | None = None,
 79        dataclass_types: dict[str, type] | list[type] | None = None,
 80        pytorch_module_types: dict[str, type] | list[type] | None = None,
 81        json_path: str = "$",
 82    ) -> None:
 83        """
 84        Args:
 85            file_path (str | Path | None, optional): Output JSON file path.
 86            artifact_path (str | Path | None, optional): Artifact path.
 87                Defaults to the parent directory of `file_path`, or a new
 88                temporary directory if `file_path` is `None`.
 89            min_artifact_size (int, optional): Byte strings (and everything
 90                that serialize to byte strings such as numpy arrays) larget
 91                than this will be stored in artifact rather than be embedded in
 92                the output JSON string/file.
 93            nodecode_types (list[str], optional): List of type names which
 94                shall be deserialized to `None` rather than their true value.
 95                See
 96                [`TB_NODECODE`](https://altaris.github.io/turbo-broccoli/turbo_broccoli.html#environment-variables)
 97            keras_format ("keras", "tf", "h5", optional): Format for Keras
 98                artifacts
 99            pandas_format ("csv", "excel", "feather", "html", "json", "latex",
100                "orc", "parquet", "pickle", "sql", "stata", "xml", optional):
101                Format for pandas artifacts
102            pandas_kwargs (dict, optional): kwargs to forward to the pandas
103                `to_*` and `read_*` function. For example, if
104                `pandas_format="parquet"`, then the content of `pandas.kwargs`
105                will be forwarded to
106                [`pandas.DataFrame.to_parquet`](https://pandas.pydata.org/docs/reference/api/pandas.DataFrame.to_parquet.html)
107                and
108                [`pandas.read_parquet`](https://pandas.pydata.org/docs/reference/api/pandas.read_parquet.html)
109            nacl_shared_key (bytes, optional): PyNaCl shared key. See also
110                [PyNaCl's
111                documentation](https://pynacl.readthedocs.io/en/latest/secret/#key)
112            dataclass_types (dict[str, type] | list[type], optional): List of
113                dataclass types for deserialization. See the
114                [README](https://altaris.github.io/turbo-broccoli/turbo_broccoli.html#supported-types).
115            pytorch_module_types (dict[str, type] | list[type], optional): List
116                of pytorch module types for deserialization. See the
117                [README](https://altaris.github.io/turbo-broccoli/turbo_broccoli.html#supported-types).
118            json_path (str, optional): Current JSONpath. Don't use.
119        """
120        self.json_path = json_path
121        self.file_path = (
122            Path(file_path) if isinstance(file_path, str) else file_path
123        )
124        if artifact_path is None:
125            if p := ENV.get("TB_ARTIFACT_PATH"):
126                self.artifact_path = Path(p)
127            else:
128                self.artifact_path = (
129                    self.file_path.parent
130                    if self.file_path is not None
131                    else Path(tempfile.mkdtemp())
132                )
133        else:
134            self.artifact_path = Path(artifact_path)
135        self.min_artifact_size = (
136            min_artifact_size
137            if min_artifact_size is not None
138            else int(ENV.get("TB_MAX_NBYTES", 8000))
139        )
140        self.nodecode_types = nodecode_types or ENV.get(
141            "TB_NODECODE", ""
142        ).split(",")
143        self.keras_format = keras_format or str(
144            ENV.get("TB_KERAS_FORMAT", "tf")
145        )
146        self.pandas_format = pandas_format or str(
147            ENV.get("TB_PANDAS_FORMAT", "csv")
148        )
149        self.pandas_kwargs = pandas_kwargs or {}
150        if isinstance(nacl_shared_key, bytes):
151            self.nacl_shared_key = nacl_shared_key
152        elif "TB_SHARED_KEY" in ENV:
153            self.nacl_shared_key = str(ENV["TB_SHARED_KEY"]).encode("utf-8")
154        else:
155            self.nacl_shared_key = None
156        self.dataclass_types = (
157            _list_of_types_to_dict(dataclass_types)
158            if isinstance(dataclass_types, list)
159            else (dataclass_types or {})
160        )
161        self.pytorch_module_types = (
162            _list_of_types_to_dict(pytorch_module_types)
163            if isinstance(pytorch_module_types, list)
164            else (pytorch_module_types or {})
165        )
166
167    def __repr__(self) -> str:
168        fp, ap = str(self.file_path), str(self.artifact_path)
169        return (
170            f"Context(file_path={fp}, artifact_path={ap}, "
171            f"json_path={self.json_path})"
172        )
173
174    def __truediv__(self, x: str | int) -> "Context":
175        """
176        Returns a copy of the current context but where the `json_path`
177        attribute is `self.json_path + "." + str(x)`. Use this when you're
178        going down the document.
179        """
180        kwargs = self.__dict__.copy()
181        kwargs["json_path"] = self.json_path + "." + str(x)
182        return Context(**kwargs)

184    def id_to_artifact_path(self, art_id: str, extension: str = "tb") -> Path:
185        """
186        Takes an artifact id (which is an UUID4 string) and returns the
187        absolute path to the corresponding artifact file.
188        """
189        art_fn = art_id + "." + extension
190        if self.file_path is not None:
191            art_fn = self.file_path.stem + "." + art_fn
192        return self.artifact_path / art_fn
193
194    def new_artifact_path(self, extension: str = "tb") -> tuple[Path, str]:
195        """Returns the path to a new artifact alongside the artifact's ID"""
196        art_id = str(uuid4())
197        return self.id_to_artifact_path(art_id, extension), art_id

199    def raise_if_nodecode(self, type_name: str) -> None:
200        """
201        Raises a `turbo_broccoli.exceptions.TypeIsNodecode` exception if
202        `type_name` or any prefix is set to not be decoded in this context (see
203        `nodecode_types` constructor argument).
204
205        For example, if `type_name` is `a.b.c`, then this method raises
206        `turbo_broccoli.exceptions.TypeIsNodecode` if either `a`, `a.b`, or
207        `a.b.c` is set as a nodecode type.
208        """
209        parts = type_name.split(".")
210        for i in range(1, len(parts) + 1):
211            t = ".".join(parts[:i])
212            if t in self.nodecode_types:
213                raise TypeIsNodecode(t)
